1ª) Crie uma tabela Aluno com as colunas nome (VARCHAR(255)), email
(VARCHAR(255)) telefone (VARCHAR(10)), altura (DECIMAL(3,2)) e aprovado
(TINYINT(1)).

CREATE TABLE alunos (
  nome VARCHAR(255),
  email VARCHAR(255),
  telefone VARCHAR(10),
  altura DECIMAL(3,2),
  aprovado TINYINT(1)
);

3ª) Utilizando a cláusula WHERE refaça os exemplos criando uma consulta para
cada tipo de operador.

Observação: nas consultas utilize valores que façam sentido de acordo com os
valores que você inseriu na tabela.

SELECT * FROM alunos WHERE nome = "Igor Brilhante";

SELECT * FROM alunos WHERE nome != "Igor Brilhante";

SELECT * FROM alunos WHERE altura <= 1.70;

SELECT * FROM alunos WHERE altura < 1.70;

SELECT * FROM alunos WHERE altura >= 1.70;

SELECT * FROM alunos WHERE altura > 1.70;

SELECT * FROM alunos WHERE altura IS NULL;

SELECT * FROM alunos WHERE altura IS NOT NULL;

SELECT * FROM alunos WHERE altura BETWEEN 1.65 AND 1.70;

SELECT * FROM alunos WHERE altura NOT BETWEEN 1.65 AND 1.70;

SELECT * FROM alunos WHERE nome LIKE 'Igor%';

SELECT * FROM alunos WHERE nome NOT LIKE 'Igor%';

SELECT * FROM alunos WHERE altura IN (1.70, 1.50);

SELECT * FROM alunos WHERE altura NOT IN (1.70, 1.50);

SELECT * FROM alunos WHERE NOT altura='1.70';
